def create
  @item = Item.new(item_params)
  if @item.save
    flash[:notice] = "Article enregistré : #{@item.name}"
    redirect_to :action=>'index'
  else
    flash[:notice] = 'Veiller remplir les champs vide!'
    render :action=>'new'
  end
end
